#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> answer;
    
    answer.reserve(arr.size() * 18);

    for (int i = 0; i < arr.size(); i++) {
        if (flag[i]) {
            answer.insert(answer.end(), arr[i] * 2, arr[i]);
        } else {
            answer.resize(answer.size() - arr[i]);
        }
    }

    return answer;
}
