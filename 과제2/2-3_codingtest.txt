#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<vector<int>> dots) {
    int minX = dots[0][0];
    int maxX = dots[0][0];
    int minY = dots[0][1];
    int maxY = dots[0][1];

    for (int i = 1; i < dots.size(); i++) {
        minX = min(minX, dots[i][0]);
        maxX = max(maxX, dots[i][0]);
        minY = min(minY, dots[i][1]);
        maxY = max(maxY, dots[i][1]);
    }

    int width = maxX - minX;
    int height = maxY - minY;

    return width * height;
}
